# derpcat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `cmd/derpcat`, a standalone netcat-like CLI that uses public Tailscale DERP for bootstrap and relay fallback, establishes a just-in-time userspace WireGuard session, and carries one bidirectional byte stream over `stdio` or a local TCP bridge.

**Architecture:** `derpcat` is split into focused `pkg/` packages: token handling, rendezvous, DERP bootstrap, traversal, userspace WireGuard, stream adapters, and session orchestration. Implementation should reach a relay-only end-to-end checkpoint first, then add direct UDP path promotion, keeping DERP available as fallback and surfacing `direct` versus `relay` path outcome clearly.

**Tech Stack:** Go 1.26.1, `mise`, `tailscale.com` packages for DERP/netcheck/disco types, `github.com/tailscale/wireguard-go`, `gvisor.dev/gvisor` netstack, Go stdlib testing.

```go
package token

import (
	"bytes"
	"encoding/base64"
	"encoding/binary"
	"errors"
	"hash/crc32"
	"time"
)

const (
	CapabilityStdio uint32 = 1 << iota
	CapabilityTCP
)

type Token struct {
	Version         uint8
	SessionID       [16]byte
	ExpiresUnix     int64
	BootstrapRegion uint16
	DERPPublic      [32]byte
	WGPublic        [32]byte
	DiscoPublic     [32]byte
	BearerSecret    [32]byte
	Capabilities    uint32
}

var (
	ErrExpired  = errors.New("token expired")
	ErrChecksum = errors.New("token checksum mismatch")
)

func Encode(tok Token) (string, error) {
	var payload bytes.Buffer
	_ = binary.Write(&payload, binary.BigEndian, tok.Version)
	payload.Write(tok.SessionID[:])
	_ = binary.Write(&payload, binary.BigEndian, tok.ExpiresUnix)
	_ = binary.Write(&payload, binary.BigEndian, tok.BootstrapRegion)
	payload.Write(tok.DERPPublic[:])
	payload.Write(tok.WGPublic[:])
	payload.Write(tok.DiscoPublic[:])
	payload.Write(tok.BearerSecret[:])
	_ = binary.Write(&payload, binary.BigEndian, tok.Capabilities)
	sum := crc32.ChecksumIEEE(payload.Bytes())
	_ = binary.Write(&payload, binary.BigEndian, sum)
	return base64.RawURLEncoding.EncodeToString(payload.Bytes()), nil
}

func Decode(encoded string, now time.Time) (Token, error) {
	var tok Token
	raw, err := base64.RawURLEncoding.DecodeString(encoded)
	if err != nil {
		return tok, err
	}
	if len(raw) < 1+16+8+2+32+32+32+32+4+4 {
		return tok, errors.New("token too short")
	}
	payload, checksumBytes := raw[:len(raw)-4], raw[len(raw)-4:]
	if got, want := crc32.ChecksumIEEE(payload), binary.BigEndian.Uint32(checksumBytes); got != want {
		return tok, ErrChecksum
	}
	r := bytes.NewReader(payload)
	_ = binary.Read(r, binary.BigEndian, &tok.Version)
	_, _ = r.Read(tok.SessionID[:])
	_ = binary.Read(r, binary.BigEndian, &tok.ExpiresUnix)
	_ = binary.Read(r, binary.BigEndian, &tok.BootstrapRegion)
	_, _ = r.Read(tok.DERPPublic[:])
	_, _ = r.Read(tok.WGPublic[:])
	_, _ = r.Read(tok.DiscoPublic[:])
	_, _ = r.Read(tok.BearerSecret[:])
	_ = binary.Read(r, binary.BigEndian, &tok.Capabilities)
	if now.Unix() > tok.ExpiresUnix {
		return tok, ErrExpired
	}
	return tok, nil
}
```

```go
package rendezvous

import (
	"crypto/hmac"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"errors"
	"sync"
	"time"

	"github.com/shayne/derpcat/pkg/token"
)

var (
	ErrClaimed = errors.New("session already claimed")
	ErrDenied  = errors.New("claim denied")
)

type Claim struct {
	Version      uint8      `json:"version"`
	SessionID    [16]byte   `json:"session_id"`
	BearerMAC    string     `json:"bearer_mac"`
	DERPPublic   [32]byte   `json:"derp_public"`
	WGPublic     [32]byte   `json:"wg_public"`
	DiscoPublic  [32]byte   `json:"disco_public"`
	Candidates   []string   `json:"candidates"`
	Capabilities uint32     `json:"capabilities"`
}

type Decision struct {
	Accepted   bool     `json:"accepted"`
	Reason     string   `json:"reason,omitempty"`
	Candidates []string `json:"candidates,omitempty"`
}

func ComputeBearerMAC(secret [32]byte, claim Claim) string {
	payload, _ := json.Marshal(struct {
		Version     uint8    `json:"version"`
		SessionID   [16]byte `json:"session_id"`
		DERPPublic  [32]byte `json:"derp_public"`
		WGPublic    [32]byte `json:"wg_public"`
		DiscoPublic [32]byte `json:"disco_public"`
	}{claim.Version, claim.SessionID, claim.DERPPublic, claim.WGPublic, claim.DiscoPublic})
	mac := hmac.New(sha256.New, secret[:])
	mac.Write(payload)
	return hex.EncodeToString(mac.Sum(nil))
}

type Gate struct {
	mu      sync.Mutex
	token   token.Token
	claimed bool
}

func NewGate(tok token.Token) *Gate { return &Gate{token: tok} }

func (g *Gate) Accept(now time.Time, claim Claim) (Decision, error) {
	g.mu.Lock()
	defer g.mu.Unlock()
	if now.Unix() > g.token.ExpiresUnix {
		return Decision{Reason: "expired"}, token.ErrExpired
	}
	if g.claimed {
		return Decision{Reason: "claimed"}, ErrClaimed
	}
	if claim.Version != g.token.Version || claim.SessionID != g.token.SessionID {
		return Decision{Reason: "mismatch"}, ErrDenied
	}
	if claim.BearerMAC != ComputeBearerMAC(g.token.BearerSecret, claim) {
		return Decision{Reason: "unauthorized"}, ErrDenied
	}
	g.claimed = true
	return Decision{Accepted: true}, nil
}
```

package stream

import (
	"context"
	"io"
	"net"
	"sync"
)

type StdioAttachment struct {
	r io.Reader
	w io.Writer
}

func NewStdioAttachment(r io.Reader, w io.Writer) *StdioAttachment {
	return &StdioAttachment{r: r, w: w}
}

func (a *StdioAttachment) Read(p []byte) (int, error)  { return a.r.Read(p) }
func (a *StdioAttachment) Write(p []byte) (int, error) { return a.w.Write(p) }

func Bridge(ctx context.Context, a, b net.Conn) error {
	errCh := make(chan error, 2)
	var once sync.Once
	copyOne := func(dst, src net.Conn) {
		_, err := io.Copy(dst, src)
		once.Do(func() {
			dst.Close()
			src.Close()
		})
		errCh <- err
	}
	go copyOne(a, b)
	go copyOne(b, a)
	select {
	case err := <-errCh:
		if err == nil || err == io.EOF {
			return nil
		}
		return err
	case <-ctx.Done():
		return ctx.Err()
	}
}
